Структуры очень похожи на массивы, но более гибкие и позволяют в себе хранить несколько пременых разных типов
(P.S Структура создаёт тип который уже хранит определённый набор переменных).

/////////////////////////////////////////////////////////////////////////////////

#include <iostream>
#include <math.h>
#include <string>
#include <windows.h>

using namespace std;


int main(int argc,char* argv[])
{
    setlocale(LC_ALL,"Russian");

    struct product{  //инфо по продукту
        int number;     //
        double weight;  // поля структуры
        string name;    //
    };

    struct worker{ //инфо по работнику
        int Age;
        int work_age;
        string FIO;
        product prod;
    };

    product a1={1,1.5,"Пачка гречки"};                      // заполнение полей структур
    worker Worker ={25,5,"Зубенко Михаил Петрович",a1};     //

    cout<<"ФИО работника "<<Worker.FIO<<"; Ответственен за "<<Worker.prod.name<<"; Весом в кг "<<a1.weight<<endl; //обращение к кокнкретным полям структур (также их можно перезаписывать)

    return 0;
}
