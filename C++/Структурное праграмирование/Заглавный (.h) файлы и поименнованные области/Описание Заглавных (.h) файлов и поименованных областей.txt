using namespace std;
extern bool prost_num(int b)
{
static double c = sqrt(b);
static bool f=true;

if(int(c)==c) {return (false); }

for (int i=2;i<c;i++)
    {
    if ((b%i)==0){f=false; return (f);}
    }

return (f);
}
