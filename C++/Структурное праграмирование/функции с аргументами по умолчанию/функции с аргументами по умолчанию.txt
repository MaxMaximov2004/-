Механизм функций с аргументами по умолчанию позволяет очень удобно задавать аргументы для функции (если для неё возможна задача аргумента по умолчанию) 

#include <iostream>
int funct(int = 1, int = 1);

int main()
{
    std::cout << funct() << "\n";
    std::cout << funct(5) << "\n";
    std::cout << funct(10,10) << "\n";
    std::cout << "Hello World!\n";
}

int funct(int a, int b) {
    return (a + b);
}
